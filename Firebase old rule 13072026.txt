rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isAdmin() {
      return request.auth != null &&
        exists(/databases/$(database)/documents/admins/$(request.auth.uid));
    }
    match /users/{uid} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && (request.auth.uid == uid || isAdmin());

      // P2P transfers only touch walletBalance — collectedCoins (mining)
      // stays locked and can only change via the user's own mine() call
      // or an admin action, never a peer transfer.
      allow update: if request.auth != null &&
        request.resource.data.diff(resource.data).affectedKeys().hasOnly(['walletBalance']);

      // Referral bonus credit at signup — only referrals + referralBonus,
      // and only exactly +1 each, so it can't be abused to inflate values
      // by more than one referral per write.
      allow update: if request.auth != null &&
        request.resource.data.diff(resource.data).affectedKeys().hasOnly(['referrals', 'referralBonus']) &&
        request.resource.data.referrals == resource.data.referrals + 1 &&
        request.resource.data.referralBonus == resource.data.referralBonus + 1;
    }
    match /usernames/{username} {
      allow read: if true;
      allow create: if request.auth != null;
    }
    match /signupOtps/{email} {
      allow read, write: if true;
    }
    match /meta/{doc} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    match /transactions/{id} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update: if isAdmin();
    }
    match /admins/{uid} {
      allow read: if request.auth != null;
      allow write: if false;
    }
  }
}
